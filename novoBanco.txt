DROP TABLE IF EXISTS cliente;
DROP TABLE IF EXISTS loja;
DROP TABLE IF EXISTS produtos;
DROP TABLE IF EXISTS complementos;


CREATE TABLE cliente (
cod INT(11) PRIMARY KEY AUTO_INCREMENT NOT NULL,
nome VARCHAR(100) NOT NULL,
numero VARCHAR(20) NOT NULL,
rua VARCHAR(100) NOT NULL,
telefone VARCHAR(10),
bairro VARCHAR(100) NOT NULL,
cidade VARCHAR(100) NOT NULL,
email VARCHAR(150) NOT NULL,
senha VARCHAR(100) NOT NULL,
complemento VARCHAR(200),
login VARCHAR(100));

CREATE TABLE loja (
nome VARCHAR(100) NOT NULL,
cod INT(11) PRIMARY KEY AUTO_INCREMENT NOT NULL,
cnpj VARCHAR(12) NOT NULL,
email VARCHAR(100),
senha VARCHAR(100),
numero VARCHAR(10),
rua VARCHAR(100),
bairro VARCHAR(100),
cidade VARCHAR(100),
complemento VARCHAR(200),
login VARCHAR(100),
status VARCHAR(10) NOT NULL,
logo VARCHAR(100) NOT NULL,
foto_fundo VARCHAR(100) NOT NULL,
descricao VARCHAR(100) NOT NULL,
dias VARCHAR(100) NOT NULL,
instagram VARCHAR(50) NOT NULL,
facebook VARCHAR(50) NOT NULL,
taxa_entrega VARCHAR(10) NOT NULL);

CREATE TABLE produtos (
cod INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
nome VARCHAR(100) NOT NULL,
descricao VARCHAR(200) NOT NULL,
preco VARCHAR(50) NOT NULL,
disponibilidade VARCHAR(20) NOT NULL,
loja INT(11) NOT NULL,
classificacao VARCHAR(50) NOT NULL);

CREATE TABLE complementos (
cod INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
nome VARCHAR(100) NOT NULL,
disponibilidade VARCHAR(15) NOT NULL,
loja INT(11) NOT NULL,
classificacao VARCHAR(50) NOT NULL);

ALTER TABLE produtos ADD CONSTRAINT produtos_loja_loja_cod FOREIGN KEY (loja) REFERENCES loja(cod);
ALTER TABLE complementos ADD CONSTRAINT complementos_loja_loja_cod FOREIGN KEY (loja) REFERENCES loja(cod);
